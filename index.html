// ==================================================== // PIP SMART TAG — Google Apps Script Backend // Version 2.0 // // SETUP: // 1. Create a Google Sheet with a tab named "Tags" // 2. Replace SPREADSHEET_ID below with your Sheet's ID // (find it in the Sheet URL: /spreadsheets/d/ID/edit) // 3. Replace SITE_URL with your Netlify domain // 4. Deploy: Extensions → Apps Script → Deploy → // New Deployment → Web App // Execute as: Me | Who can access: Anyone // 5. Copy the deployment URL into index.html (API const) // ==================================================== const SPREADSHEET_ID = 'YOUR_SPREADSHEET_ID_HERE'; const SHEET_NAME = 'Tags'; const SITE_URL = 'https://YOUR_NETLIFY_DOMAIN_HERE'; // Column positions (1-indexed) const C = { TAG_ID: 1, TOKEN: 2, OWNER_NAME: 3, OWNER_EMAIL: 4, OWNER_PHONE: 5, OWNER_WHATSAPP:6, DOG_NAME: 7, BREED: 8, PHOTO_URL: 9, LIKES: 10, QUIRKS: 11, BIRTHDAY: 12, INSTAGRAM: 13, ADDRESS: 14, MICROCHIP: 15, VET_STATUS: 16, VET_NOTES: 17, VET_NAME: 18, VET_PHONE: 19, REGISTERED: 20, LAST_UPDATED: 21, }; // ==================================================== // SHEET ACCESS // ==================================================== function getSheet() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { // Auto-create the sheet with headers if it doesn't exist sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ 'Tag ID', 'Token', 'Owner Name', 'Owner Email', 'Owner Phone', 'Owner WhatsApp', 'Dog Name', 'Breed', 'Photo URL', 'Likes', 'Quirks', 'Birthday', 'Instagram', 'Address', 'Microchip', 'Vet Status', 'Vet Notes', 'Vet Name', 'Vet Phone', 'Registered', 'Last Updated' ]); sheet.setFrozenRows(1); } return sheet; } function findRow(sheet, tagId) { if (!tagId) return null; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]).toLowerCase() === String(tagId).toLowerCase()) { return { rowNum: i + 1, row: data[i] }; } } return null; } function generateToken() { const chars = 'ABCDEFGHJKMNPQRSTUVWXYZabcdefghjkmnpqrstuvwxyz23456789'; let t = ''; for (let i = 0; i < 28; i++) t += chars[Math.floor(Math.random() * chars.length)]; return t; } // ==================================================== // GET HANDLER — serves profile data and verify requests // ==================================================== function doGet(e) { try { const p = e.parameter; if (p.action === 'get') return respond(handleGet(p.tagId)); if (p.action === 'verify') return respond(handleVerify(p.tagId, p.token)); return respond({ error: 'Unknown action' }); } catch (err) { Logger.log(err); return respond({ error: err.message }); } } // ==================================================== // POST HANDLER // ==================================================== function doPost(e) { try { const body = JSON.parse(e.postData.contents); if (body.action === 'register') return respond(handleRegister(body)); if (body.action === 'update') return respond(handleUpdate(body)); if (body.action === 'location') return respond(handleLocation(body)); if (body.action === 'found') return respond(handleFound(body)); return respond({ error: 'Unknown action' }); } catch (err) { Logger.log(err); return respond({ error: err.message }); } } // ==================================================== // GET PROFILE // ==================================================== function handleGet(tagId) { if (!tagId) return { error: 'No tagId provided' }; const sheet = getSheet(); const found = findRow(sheet, tagId); if (!found) return { error: 'Tag not found' }; return rowToProfile(found.row); } function rowToProfile(r) { return { dogName: r[C.DOG_NAME - 1], breed: r[C.BREED - 1], photoUrl: r[C.PHOTO_URL - 1], likes: r[C.LIKES - 1], quirks: r[C.QUIRKS - 1], birthday: formatDateForApi(r[C.BIRTHDAY - 1]), instagram: r[C.INSTAGRAM - 1], address: r[C.ADDRESS - 1], microchip: r[C.MICROCHIP - 1], vetStatus: r[C.VET_STATUS - 1], vetNotes: r[C.VET_NOTES - 1], vetName: r[C.VET_NAME - 1], vetPhone: r[C.VET_PHONE - 1], ownerName: r[C.OWNER_NAME - 1], ownerPhone: r[C.OWNER_PHONE - 1], ownerWhatsapp: r[C.OWNER_WHATSAPP - 1], // Note: ownerEmail and token are NEVER returned publicly }; } // ==================================================== // VERIFY TOKEN (for edit form pre-fill) // ==================================================== function handleVerify(tagId, token) { if (!tagId || !token) return { error: 'Missing tagId or token' }; const sheet = getSheet(); const found = findRow(sheet, tagId); if (!found) return { error: 'Tag not found' }; if (String(found.row[C.TOKEN - 1]) !== String(token)) return { error: 'Invalid token' }; // Return profile data for pre-filling the edit form return rowToProfile(found.row); } // ==================================================== // REGISTER NEW TAG // ==================================================== function handleRegister(body) { const { tagId, ownerEmail, dogName } = body; if (!tagId) return { error: 'Missing tagId' }; if (!ownerEmail) return { error: 'Missing owner email' }; if (!dogName) return { error: 'Missing dog name' }; const sheet = getSheet(); if (findRow(sheet, tagId)) return { error: 'This tag has already been registered.' }; const token = generateToken(); const now = new Date(); sheet.appendRow([ tagId, token, body.ownerName || '', ownerEmail, body.ownerPhone || '', body.ownerWhatsapp || body.ownerPhone || '', dogName, body.breed || '', body.photoUrl || '', body.likes || '', body.quirks || '', body.birthday || '', body.instagram || '', body.address || '', body.microchip || '', body.vetStatus || '', body.vetNotes || '', body.vetName || '', body.vetPhone || '', now, now, ]); const editUrl = `${SITE_URL}/?tag=${tagId}&token=${token}`; sendEditEmail(ownerEmail, body.ownerName || 'there', dogName, editUrl, tagId); return { success: true, token }; } // ==================================================== // UPDATE EXISTING TAG // ==================================================== function handleUpdate(body) { const { tagId, token, data } = body; if (!tagId || !token) return { error: 'Missing tagId or token' }; const sheet = getSheet(); const found = findRow(sheet, tagId); if (!found) return { error: 'Tag not found' }; if (String(found.row[C.TOKEN - 1]) !== String(token)) return { error: 'Invalid token' }; const row = found.rowNum; const d = data || {}; function set(col, v) { sheet.getRange(row, col).setValue(v != null ? v : ''); } set(C.DOG_NAME, d.dogName); set(C.BREED, d.breed); set(C.PHOTO_URL, d.photoUrl); set(C.LIKES, d.likes); set(C.QUIRKS, d.quirks); set(C.BIRTHDAY, d.birthday); set(C.INSTAGRAM, d.instagram); set(C.OWNER_NAME, d.ownerName); set(C.OWNER_PHONE, d.ownerPhone); set(C.OWNER_WHATSAPP, d.ownerWhatsapp || d.ownerPhone); set(C.ADDRESS, d.address); set(C.MICROCHIP, d.microchip); set(C.VET_STATUS, d.vetStatus); set(C.VET_NOTES, d.vetNotes); set(C.VET_NAME, d.vetName); set(C.VET_PHONE, d.vetPhone); set(C.LAST_UPDATED, new Date()); return { success: true }; } // ==================================================== // LOCATION ALERT // ==================================================== function handleLocation(body) { const { tagId, lat, lon, mapsUrl } = body; if (!tagId) return { error: 'Missing tagId' }; const sheet = getSheet(); const found = findRow(sheet, tagId); if (!found) return { success: true }; // silent — don't expose whether tag exists const r = found.row; const email = r[C.OWNER_EMAIL - 1]; const ownerName = r[C.OWNER_NAME - 1] || 'there'; const dogName = r[C.DOG_NAME - 1] || 'your dog'; const mapLink = mapsUrl || `https://maps.google.com/maps?q=${lat},${lon}`; if (email) { try { MailApp.sendEmail({ to: email, subject: `📍 ${dogName} has been found — location inside`, htmlBody: `

Someone found ${dogName}! 🐾

A finder shared their location using ${dogName}'s Pip Smart Tag.

📍 See Where They Are

Coordinates: ${lat}, ${lon}

Head there now — ${dogName} is waiting! 🚗

Pip Smart Tags — No matter where they roam, Pip will find the way home.

` }); } catch (err) { Logger.log('Email error: ' + err.message); } } return { success: true }; } // ==================================================== // FOUND NOTIFICATION // ==================================================== function handleFound(body) { const { tagId } = body; if (!tagId) return { error: 'Missing tagId' }; const sheet = getSheet(); const found = findRow(sheet, tagId); if (!found) return { success: true }; const r = found.row; const email = r[C.OWNER_EMAIL - 1]; const dogName = r[C.DOG_NAME - 1] || 'your dog'; if (email) { try { MailApp.sendEmail({ to: email, subject: `✅ ${dogName} is safe — marked as found`, htmlBody: `

✅ ${dogName} is safe!

Someone scanned ${dogName}'s Pip Smart Tag and marked them as safe.

Check your phone for any messages or calls from the finder.

Pip Smart Tags — No matter where they roam, Pip will find the way home.

` }); } catch (err) { Logger.log('Email error: ' + err.message); } } return { success: true }; } // ==================================================== // SEND EDIT LINK EMAIL (on registration) // ==================================================== function sendEditEmail(to, ownerName, dogName, editUrl, tagId) { const tagUrl = `${SITE_URL}/?tag=${tagId}`; try { MailApp.sendEmail({ to, subject: `🐾 ${dogName}'s Pip Smart Tag is live!`, htmlBody: `

Hi ${ownerName}! ${dogName}'s tag is ready 🎉

${dogName}'s Pip Smart Tag is live. If they're ever lost, anyone who scans the tag will see their profile and be able to contact you instantly.

🔐 Your Private Edit Link

Use this anytime to update ${dogName}'s details — new vet, new address, new photo. Keep this email.

✏️ Edit ${dogName}'s Tag

Or copy this URL: ${editUrl}

🔗 Your Tag's Public Link

This is the link anyone sees when they scan the tag:

${tagUrl}

⚠️ Keep your edit link private.
Anyone with that link can change your dog's details. Don't share it.

Pip Smart Tags — No matter where they roam, Pip will find the way home.

` }); } catch (err) { Logger.log('sendEditEmail error: ' + err.message); } } // ==================================================== // UTILITIES // ==================================================== function respond(data) { const output = ContentService.createTextOutput(JSON.stringify(data)); output.setMimeType(ContentService.MimeType.JSON); return output; } function formatDateForApi(val) { if (!val) return ''; try { if (val instanceof Date) { return Utilities.formatDate(val, 'UTC', 'yyyy-MM-dd'); } return String(val); } catch { return ''; } }
